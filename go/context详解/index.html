    "context"
    "fmt"

    gen := func(ctx context.Context) <-chan int {

        n := 1
        go func() {
            for {
                select {
                case <-ctx.Done():
                    return // returning not to leak the goroutine
                case dst <- n:
                    n++
                }
            }

    }
